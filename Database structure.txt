DROP TABLE INTERFACES;
CREATE TABLE INTERFACES
( ID INTEGER PRIMARY KEY AUTOINCREMENT,
ROUTER_NAME TEXT  NOT NULL,
NAME TEXT NOT NULL,
DESCRIPTION TEXT,
BW INT,
VLAN_TAG_EXT INT,
VLAN_TAG_INT INT,
VRF TEXT,
IP_ADDRESS INT NOT NULL,
IP_MASK INT NOT NULL ,
QOS_IN TEXT,
QOS_OUT TEXT,
HSRP_GROUP INT,
HSRP_PRIORITY INT,
SHUT BOOLEAN NOT NULL
);

DROP TABLE BGP;
CREATE TABLE BGP 
( ID INTEGER PRIMARY KEY AUTOINCREMENT,
ROUTER_NAME TEXT NOT NULL,
PEER_GROUP TEXT,
VRF TEXT,
IS_PEER_GROUP BOOLEAN NOT NULL,
IP INT,
ASN INT,
DESCRIPTION TEXT,
PL_IN TEXT,
PL_OUT TEXT,
RMAP_IN TEXT,
RMAP_OUT TEXT,
SHUT BOOLEAN NOT NULL);



DROP TABLE STATIC;
CREATE TABLE STATIC
(ID INTEGER PRIMARY KEY AUTOINCREMENT,
ROUTER_NAME TEXT NOT NULL,
VRF TEXT,
IP_ADDRESS INT NOT NULL,
IP_MASK INT NOT NULL ,
NEXT_HOP INT NOT NULL,
NAME TEXT);
